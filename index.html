import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt
import random

# Seed for reproducibility
np.random.seed(42)
random.seed(42)

# Possible categories
departments = ["R&D", "Sales", "IT", "HR", "Finance", "Marketing", "Operations"]
regions = ["Europe", "Asia", "Middle East", "Africa", "North America", "South America"]

# Generate 100 rows of data
data = []
for i in range(1, 101):
    employee_id = f"EMP{i:03d}"
    department = random.choice(departments)
    region = random.choice(regions)
    performance_score = round(np.random.uniform(50, 95), 2)  # between 50 and 95
    years_experience = np.random.randint(1, 21)  # between 1 and 20 years
    satisfaction_rating = round(np.random.uniform(2.5, 5.0), 1)  # between 2.5 and 5.0
    data.append([employee_id, department, region, performance_score, years_experience, satisfaction_rating])

# Create DataFrame
df = pd.DataFrame(data, columns=["employee_id", "department", "region", "performance_score", "years_experience", "satisfaction_rating"])

# 1. Frequency count for Finance department
finance_count = (df["department"] == "Finance").sum()
print(f"Frequency count for Finance department: {finance_count}")

# 2. Create histogram of departments
plt.figure(figsize=(8,6))
sns.countplot(data=df, x="department", palette="Set2", order=df["department"].value_counts().index)
plt.title("Distribution of Departments", fontsize=14)
plt.xlabel("Department")
plt.ylabel("Count")
plt.xticks(rotation=30)
plt.tight_layout()
plt.savefig("histogram.png", dpi=300)
plt.close()
